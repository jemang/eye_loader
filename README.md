# eye_loader

1. use system shell
    - sudo apt install ruby-full ruby-dev
    - rbenv shell system

2. install eye global
    - sudo gem install eye
    - sudo apt install ruby-dev (if errors install eye)

3. load eye conf
    - /usr/local/bin/eye l config/eye.rb

4. make alias
    - alias eye='/usr/local/bin/eye'
